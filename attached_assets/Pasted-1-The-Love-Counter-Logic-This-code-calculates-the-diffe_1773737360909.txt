1. The Love Counter Logic
This code calculates the difference between today and your special dates.

TypeScript
const LoveCounter = () => {
  const engagementDate = new Date('2024-10-12');
  const marriageDate = new Date('2025-07-20');
  const today = new Date();

  const diffDays = (date1: Date, date2: Date) => {
    const diffTime = Math.abs(date2.getTime() - date1.getTime());
    return Math.ceil(diffTime / (1000 * 60 * 60 * 24));
  };

  const daysSinceEngagement = diffDays(engagementDate, today);
  const daysSinceMarriage = diffDays(marriageDate, today);

  return (
    <div className="grid grid-cols-2 gap-4 w-full">
      {/* Engagement Card */}
      <div className="p-4 rounded-2xl bg-blue-500/10 border border-blue-500/20 backdrop-blur-md text-center">
        <p className="text-[10px] uppercase tracking-widest text-[#8ab4f8] mb-1">Since Engagement</p>
        <div className="text-2xl font-bold text-white leading-none">
          {daysSinceEngagement} <span className="text-xs font-normal opacity-60">Days</span>
        </div>
        <p className="text-[9px] text-white/40 mt-1">Oct 12, 2024</p>
      </div>

      {/* Marriage Card */}
      <div className="p-4 rounded-2xl bg-pink-500/10 border border-pink-500/20 backdrop-blur-md text-center">
        <p className="text-[10px] uppercase tracking-widest text-pink-300 mb-1">Since Marriage</p>
        <div className="text-2xl font-bold text-white leading-none">
          {daysSinceMarriage} <span className="text-xs font-normal opacity-60">Days</span>
        </div>
        <p className="text-[9px] text-white/40 mt-1">July 20, 2025</p>
      </div>
    </div>
  );
};
